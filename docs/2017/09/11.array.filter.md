数组过滤
===

数组过滤：
```js
function arrayFilter(array) {
  var array = array || [];
  var obj = {};

  for (var i = 0, j = array.length; i < j; i++) {
    var v = array[i];

    if (typeof (obj[v]) == 'undefined') {
      obj[v] = 1;
    }
  }

  array.length = 0;

  for (var i in obj) {
    array[array.length] = i;
  }

  return array;
}
```
